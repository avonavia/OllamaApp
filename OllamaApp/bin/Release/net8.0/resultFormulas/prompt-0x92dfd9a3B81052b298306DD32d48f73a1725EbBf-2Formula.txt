// This method corresponds to the sub function from the Solidity contract
    // It performs a subtraction of two uint256 values, throwing an exception if the second value is greater than the first one
    private BigDecimal Sub(BigDecimal a, BigDecimal b, string errorMessage)
    {
        // Check if b is less than or equal to a, otherwise throw an exception with the provided error message
        if (b > a)
            throw new Exception(errorMessage);

        // Perform the subtraction and return the result
        BigDecimal c = a - b;
        return c;
    }
