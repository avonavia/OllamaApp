```csharp
[assembly: FormulaAssembly("getReserves")]

namespace Sauron.Formula.getReserves
{
    public class getReservesFormula : IFormulaPlugin
    {
        public CalculationResult<IEnumerable<KeyValuePair<string, BigDecimal>>?> Formula(TransactionInfo info,
            IEnumerable<KeyValuePair<string, decimal>>? tokenPrices = null)
        {
            var reserve0 = new KeyValuePair<string, BigDecimal>("reserve0", new BigDecimal(info.Reserve0));
            var reserve1 = new KeyValuePair<string, BigDecimal>("reserve1", new BigDecimal(info.Reserve1));
            var blockTimestampLast = new KeyValuePair<string, BigDecimal>("blockTimestampLast", new BigDecimal((ulong)info.BlockTimestampLast));

            return CalculationResult<IEnumerable<KeyValuePair<string, BigDecimal>>?>.Success(new List<KeyValuePair<string, BigDecimal>> { reserve0, reserve1, blockTimestampLast });
        }
    }
}
```
