namespace Sauron.Formula.UQDIV
{
    public class UQDIVFormula : IFormulaPlugin
    {
        public CalculationResult<BigDecimal> Formula(TransactionInfo info, BigDecimal x, BigDecimal y)
        {
            // Perform unsigned integer division and return the result as a BigDecimal.
            var z = x / y;

            // Return the calculation result without any message for the user.
            return new CalculationResult<BigDecimal>(z);
        }
    }
}
