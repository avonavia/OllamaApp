// Method to handle approval of tokens for spending by another address
    private void HandleApproval(string owner, string spender, BigDecimal amount)
    {
        // Check if the owner is not the zero address
        if (owner == "0x0")
            throw new Exception("ERC20: approve from the zero address");

        // Check if the spender is not the zero address
        if (spender == "0x0")
            throw new Exception("ERC20: approve to the zero address");

        // Update the allowance for the owner-spender pair with the approved amount
        Allowances[owner][spender] = amount;
    }
