```csharp
using System;

namespace PNLCalculator
{
    public class Calculator
    {
        /// <summary>
        /// Calculate PNL for user addresses.
        /// </summary>
        /// <param name="amount0Out">Amount of token 0 output.</param>
        /// <param name="amount1Out">Amount of token 1 output.</param>
        /// <param name="reserve0">Current reserve of token 0 in the pool.</param>
        /// <param name="reserve1">Current reserve of token 1 in the pool.</param>
        /// <param name="balance0">User's balance of token 0 before the swap.</param>
        /// <param name="balance1">User's balance of token 1 before the swap.</param>
        /// <returns>PNL value in Eth.</returns>
        public decimal CalculatePNL(decimal amount0Out, decimal amount1Out, decimal reserve0, decimal reserve1, decimal balance0, decimal balance1)
        {
            // Calculate amount of tokens user got from the pool.
            var amount0In = (balance0 > reserve0 - amount0Out) ? balance0 - (reserve0 - amount0Out) : 0;
            var amount1In = (balance1 > reserve1 - amount1Out) ? balance1 - (reserve1 - amount1Out) : 0;

            // Calculate user's final token balances.
            var finalBalance0 = balance0 - amount0Out + amount0In;
            var finalBalance1 = balance1 - amount1Out + amount1In;

            // Calculate initial and final values of tokens in Eth.
            // Assuming the price of each token is 1 Eth for simplicity.
            // In a real-world scenario, you should use actual token prices.
            var initialValue = balance0 + balance1;
            var finalValue = finalBalance0 + finalBalance1;

            // Calculate PNL.
            var pnl = finalValue - initialValue;

            return pnl;
        }
    }
}
```
