// The PNL calculation method for UniswapV2 swap transaction
using System;

public class UniswapPNLCalculator
{
    // This method calculates Profit & Loss (PNL) for a user in Eth from a given Swap transaction
    public static decimal CalculatePNL(decimal amount0Out, decimal amount1Out, decimal reserve0, decimal reserve1, decimal price0, decimal price1, string fromAddress, string toAddress)
    {
        // Calculate the cost of tokens sent out in Eth
        decimal costOfTokensSent = amount0Out * price0 + amount1Out * price1;

        // If user sends money to themselves, no fee is deducted
        if (fromAddress == toAddress)
            return -costOfTokensSent;

        // Calculate the new reserve after swap
        decimal newReserve0 = reserve0 + amount0Out;
        decimal newReserve1 = reserve1 + amount1Out;

        // Calculate the new price after swap
        decimal newPrice0 = newReserve1 / newReserve0;
        decimal newPrice1 = newReserve0 / newReserve1;

        // Calculate the value of tokens received in Eth
        decimal valueOfTokensReceived = amount0Out * newPrice0 + amount1Out * newPrice1;

        // Calculate PNL (assuming fees are not refunded)
        decimal pnl = valueOfTokensReceived - costOfTokensSent;

        return pnl;
    }
}
