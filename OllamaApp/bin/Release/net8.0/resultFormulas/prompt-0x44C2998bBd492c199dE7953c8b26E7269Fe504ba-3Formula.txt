// This method performs multiplication of two numbers and handles overflow by checking if division results in the original number
    public BigDecimal Mul(BigDecimal a, BigDecimal b)
    {
        if (a == 0)
            return 0;

        BigDecimal c = a * b;
        // Check for multiplication overflow. If the result of dividing c by a is not equal to b, then an overflow has occurred
        if (c / a != b)
            throw new ArgumentException("SafeMath: multiplication overflow");

        return c;
    }
