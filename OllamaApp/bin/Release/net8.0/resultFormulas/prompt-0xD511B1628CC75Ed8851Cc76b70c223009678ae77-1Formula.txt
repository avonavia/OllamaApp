//Method to add two numbers and handle overflow
    public CalculationResult<BigInteger> Add(BigInteger x, BigInteger y)
    {
        try
        {
            var z = checked(x + y); //Checked keyword will throw an OverflowException if addition overflows
            return new CalculationResult<BigInteger>(z);
        }
        catch (OverflowException ex)
        {
            return new CalculationResult<BigInteger>("ds-math-add-overflow", null, ex.Message);
        }
    }
