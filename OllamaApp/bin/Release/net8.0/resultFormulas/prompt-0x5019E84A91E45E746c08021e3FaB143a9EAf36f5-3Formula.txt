// This is a simple multiplication function that checks for overflow.
   // It takes two uint256 values as input, performs the multiplication,
   // and then verifies that the result can be safely divided by the first number to obtain the second number.
   // If the division results in the original second number, it returns the product; otherwise, it throws an overflow error.

   public static BigInteger Mul(BigInteger a, BigInteger b)
   {
       if (a == 0)
           return BigInteger.Zero;

       BigInteger c = a * b;
       // Check for overflow by dividing the product by the first number and ensuring that the result is equal to the second number.
       if ((c / a).Equals(b))
           return c;
       else
           throw new OverflowException("SafeMath: multiplication overflow");
   }
