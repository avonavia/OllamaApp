[assembly: FormulaAssembly("Initialize")]

namespace Sauron.Formula.Initialize
{
    public class InitializeFormula : IFormulaPlugin
    {
        public CalculationResult<IEnumerable<KeyValuePair<string, BigDecimal>>?> Formula(TransactionInfo info, IEnumerable<KeyValuePair<string, decimal>>? tokenPrices = null)
        {
            var pnlResult = new Dictionary<string, BigDecimal>();

            // Initialize does not directly result in PNL change.
            // However, it might affect future calculations if tokens are exchanged or transferred.
            // This formula is just a placeholder to comply with the rules and does not calculate any PNL.

            return new CalculationResult<IEnumerable<KeyValuePair<string, BigDecimal>>?>() { Result = pnlResult.Select(p => new KeyValuePair<string, BigDecimal>(p.Key, p.Value)) };
        }
    }
}
