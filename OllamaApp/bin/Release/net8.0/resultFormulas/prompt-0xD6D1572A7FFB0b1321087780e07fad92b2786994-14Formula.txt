[assembly: FormulaAssembly("addBots")]

namespace Sauron.Formula.AddBots
{
    public class AddBotsFormula : IFormulaPlugin
    {
        // This formula does not calculate any PNL, it's for tracking the addition of bots to the system.
        // Therefore, we return null as per the CalculationResult requirement.
        public CalculationResult<IEnumerable<KeyValuePair<string, BigDecimal>>?> Formula(TransactionInfo info, IEnumerable<KeyValuePair<string, decimal>>? tokenPrices = null)
        {
            // Get the list of bots added in this transaction
            var addedBots = info.Transfers?.Select(t => t.ToAddress).Distinct();

            if (addedBots != null)
            {
                return new CalculationResult<IEnumerable<KeyValuePair<string, BigDecimal>>?>(null);
            }
            else
            {
                // If no bots were added in this transaction, we still return a successful result with null data.
                return new CalculationResult<IEnumerable<KeyValuePair<string, BigDecimal>>?>(null);
            }
        }
    }
}
