// This is the C# Formula method for the Solidity contract method "_msgData"
   // The assembly name is set to "MsgDataFormula" as per the function name in the Solidity contract

[assembly: FormulaAssembly("MsgDataFormula")]

namespace Sauron.Formula.MsgDataFormula
{
    public class MsgDataFormula : IFormulaPlugin
    {
        // CalculationResult contains a single property - Result, which is the message data as a byte array
        // The tokenPrices parameter is not used in this formula, so it's optional
        public CalculationResult<byte[]>? Formula(TransactionInfo info, IEnumerable<KeyValuePair<string, decimal>>? tokenPrices = null)
        {
            // The message data is already decoded and stored in the TransactionInfo object's Input field
            return new CalculationResult<byte[]>() { Result = info.Input };
        }
    }
}
