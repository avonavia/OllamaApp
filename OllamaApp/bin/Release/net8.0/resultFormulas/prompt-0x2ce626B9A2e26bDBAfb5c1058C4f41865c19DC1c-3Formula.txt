[assembly: FormulaAssembly("Initialize")]

namespace Sauron.Formula.Initialize

public class InitializeFormula : IFormulaPlugin
{
    public CalculationResult<IEnumerable<KeyValuePair<string, BigDecimal>>?> Formula(TransactionInfo info, IEnumerable<KeyValuePair<string, decimal>>? tokenPrices = null)
    {
        // This formula does not calculate PNL for the initialize function of UniswapV2 contract as it doesn't involve any transfers that could affect a user's balance.
        // Therefore, we return null to indicate that there is no PNL to calculate.
        return CalculationResult<IEnumerable<KeyValuePair<string, BigDecimal>>?>.Success(null);
    }
}
