//Method to get PNL for an approve transaction
    public CalculationResult<IEnumerable<KeyValuePair<string, BigDecimal>>?> Formula(TransactionInfo info, IEnumerable<KeyValuePair<string, decimal>>? tokenPrices = null)
    {
        var sender = info.FromAddress;
        var spender = info.ToAddress;

        //Approve transaction does not result in any direct PNL for the sender or spender
        return new CalculationResult<IEnumerable<KeyValuePair<string, BigDecimal>>?>
        {
            Result = null, //No direct PNL change
            Message = $"Approve transaction from {sender} to {spender}"
        };
    }
