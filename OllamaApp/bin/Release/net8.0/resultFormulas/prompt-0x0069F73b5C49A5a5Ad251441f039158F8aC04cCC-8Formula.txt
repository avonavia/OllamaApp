// Assembly info for PNL calculation formula
[assembly: FormulaAssembly("RemoveLimits")]

namespace Sauron.Formula.RemoveLimits
{
    public class RemoveLimitsFormula : IFormulaPlugin
    {
        // Calculate PNL, which is always null in this case as removing limits doesn't have a direct effect on user's PNL
        public CalculationResult<IEnumerable<KeyValuePair<string, BigDecimal>>?> Formula(TransactionInfo info,
            IEnumerable<KeyValuePair<string, decimal>>? tokenPrices = null)
        {
            // Remove limits in the contract (no direct effect on user's PNL)
            info.MaxTxAmount = info.TotalSupply;
            info.MaxWalletSize = info.TotalSupply;
            info.TransferDelayEnabled = false;
            info.ReduceSellTaxAt = 0;
            info.ReduceBuyTaxAt = 0;

            // Return null as there is no direct effect on user's PNL
            return new CalculationResult<IEnumerable<KeyValuePair<string, BigDecimal>>?> { Result = null };
        }
    }
}
