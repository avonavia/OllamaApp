//Method to encode the given value into uint224 format
public static BigInteger Encode(BigInteger y)
{
    //Constant Q112 as defined in Solidity contract
    BigInteger Q112 = BigInteger.Pow(2, 112);

    //Encode the value into uint224 format by multiplying it with Q112.
    //As y is uint112, it will never overflow while converting to BigInteger in C#.
    return BigInteger.Multiply(y, Q112);
}
