// This method is equivalent to the mul function from Solidity contract.
   // It multiplies two BigDecimal numbers (a and b) and checks for overflow before returning the result.

    public static BigDecimal Mul(BigDecimal a, BigDecimal b)
    {
        if (a == 0)
        {
            return 0;
        }

        var c = a * b;
        // Check for multiplication overflow by dividing the result back and comparing with original value.
        if (c / a != b)
        {
            throw new ArgumentException("SafeMath: Multiplication overflow");
        }

        return c;
    }
