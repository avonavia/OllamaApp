// This method does not calculate PNL directly as it is not a financial transaction, but it updates the state of 'bots' dictionary.
// Since we don't have any data about financial transactions related to 'bots', this formula will always return null.

[assembly: FormulaAssembly("delBots")]

namespace Sauron.Formula.delBots
{
    public class delBotsFormula : IFormulaPlugin
    {
        public CalculationResult<IEnumerable<KeyValuePair<string, BigDecimal>>?> Formula(TransactionInfo info,
            IEnumerable<KeyValuePair<string, decimal>>? tokenPrices = null)
        {
            // As the method only updates 'bots' dictionary, there is no direct PNL calculation.
            return CalculationResult<IEnumerable<KeyValuePair<string, BigDecimal>>>.Success(null);
        }
    }
}
