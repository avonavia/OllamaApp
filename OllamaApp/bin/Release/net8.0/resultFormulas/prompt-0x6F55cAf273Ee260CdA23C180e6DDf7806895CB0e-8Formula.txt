using System;

public class PnlCalculator
{
    // Assembly info for the PNL calculation formula
    [System.Reflection.AssemblyDescription("PNL Calculation Formula for UniswapV2 Swap Method")]

    /// <summary>
    /// Calculate the profit and loss (PNL) for a given swap transaction in Ethereum.
    /// </summary>
    /// <param name="amount0Out">The amount of token0 sent to the recipient.</param>
    /// <param name="amount1Out">The amount of token1 sent to the recipient.</param>
    /// <param name="reserve0">The reserve amount of token0 in the contract before the swap.</param>
    /// <param name="reserve1">The reserve amount of token1 in the contract before the swap.</param>
    /// <param name="balance0After">The balance amount of token0 in the contract after the swap.</param>
    /// <param name="balance1After">The balance amount of token1 in the contract after the swap.</param>
    /// <param name="feePercentage">The fee percentage that was deducted from the amounts sent to the recipient (in decimal).</param>
    /// <returns>Profit and loss (PNL) for the user in Ethereum.</returns>
    public static decimal CalculatePnl(decimal amount0Out, decimal amount1Out, decimal reserve0, decimal reserve1, decimal balance0After, decimal balance1After, decimal feePercentage)
    {
        // Calculate the amounts sent to the recipient (excluding fees)
        var amount0Sent = amount0Out / (1 - feePercentage);
        var amount1Sent = amount1Out / (1 - feePercentage);

        // Calculate the amounts swapped into the contract
        var amount0In = Math.Max(balance0After - (reserve0 - amount0Out), 0);
        var amount1In = Math.Max(balance1After - (reserve1 - amount1Out), 0);

        // Calculate the value of tokens swapped into the contract in Ethereum
        var amount0InEth = amount0In / reserve0 * (reserve0 + reserve1);
        var amount1InEth = amount1In / reserve1 * (reserve0 + reserve1);

        // Calculate the total value of tokens sent to the recipient in Ethereum
        var totalSentEth = amount0Sent / reserve0 * (reserve0 + reserve1) + amount1Sent / reserve1 * (reserve0 + reserve1);

        // Calculate PNL and return it in Ethereum
        return -totalSentEth + amount0InEth + amount1InEth;
    }
}
