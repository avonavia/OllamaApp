public class MathFormula
{
    public uint Sqrt(uint y)
    {
        uint z;
        if (y > 3)
        {
            z = y;
            uint x = y / 2 + 1;
            while (x < z)
            {
                z = x;
                x = (y / x + x) / 2;
            }
        }
        else if (y != 0)
        {
            z = 1;
        }
        else
        {
            z = 0;
        }
        return z;
    }
}
