[assembly: FormulaAssembly("Min")]

namespace Sauron.Formula.Min
{
    public class MinFormula : IFormulaPlugin
    {
        public CalculationResult<BigDecimal> Formula(TransactionInfo info, IEnumerable<KeyValuePair<string, decimal>>? tokenPrices = null)
        {
            // Extract the two arguments from the TransactionInfo object
            BigDecimal x = info.Arguments[0];
            BigDecimal y = info.Arguments[1];

            // Calculate the minimum of x and y
            BigDecimal z = (x < y) ? x : y;

            // Return the result as a CalculationResult object
            return new CalculationResult<BigDecimal>(z, null);
        }
    }
}
